function calculateSmartRank(userInput, itemName) { var query = userInput.trim().toLowerCase(); var name = itemName.toLowerCase(); var cleanQuery = query.replace(/[\s\+\-\/\(\)]+/g, ''); var cleanName = name.replace(/[\s\+\-\/\(\)]+/g, ''); // --- 第一層：連續包含比對 (權重最高) --- // 如果品名裡有連續的 "高絲bb"，這筆一定要排第一 if (cleanName.indexOf(cleanQuery) !== -1) { return { score: 200 - (cleanName.length * 0.1), ratio: 1.0 }; } // --- 第二層：分段塊比對 (英文塊必須完整) --- var segments = query.match(/[\u4E00-\u9FA5]+|[a-z0-9]+/g); if (segments && segments.length > 1) { var allMatch = true; var blockScore = 0; for (var i = 0; i < segments.length; i++) { var seg = segments[i]; // 檢查該區段是否在品名中「連著出現」 if (cleanName.indexOf(seg) === -1) { allMatch = false; break; } blockScore += (seg.length); } // 如果 segments 裡包含英文 (如 BB)，且品名中確實有連著的 BB if (allMatch) { var finalRatio = blockScore / cleanQuery.length; // 給予 150 分以上的權重，確保高過單純的字元命中 return { score: 150 + (finalRatio * 10), ratio: finalRatio }; } } // --- 第三層：弱命中 (字元命中) --- // 像 Biore 這種只有「高」跟「絲」分開出現的，分數會被壓低 var matchCount = 0; for (var c = 0; c < cleanQuery.length; c++) { if (cleanName.indexOf(cleanQuery.charAt(c)) !== -1) matchCount++; } var ratio = matchCount / cleanQuery.length; // 只有命中率夠高且不符合前兩項時才顯示，分數壓在 50 以下 if (ratio >= 0.8) { return { score: 50 * ratio, ratio: ratio }; } return { score: 0, ratio: 0 }; }
